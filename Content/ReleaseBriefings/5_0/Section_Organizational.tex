\begin{frame}[fragile]{Kokkos Training \& Hackathon @ CEA}
  \begin{center}
    \textbf{Kokkos Training \& Hackathon @ CEA, France}
  \end{center}

  \begin{itemize}
    \item \textit{When:} January 12th-16th 2026 (noon to noon)
    \item \textit{Where:} CEA, Saclay (``near'' Paris), France
    \item \textit{What:}
      \begin{itemize}
        \item 1 day of basic training: masterclass + exercises
        \item 3 days Hackathon to get your hands dirty \& discover Kokkos for real:
        \begin{itemize}
          \item get a code half-ported to GPU,
          \item in small teams, with support from the devs., finish porting \& optimize for GPU,
          \item compete with the other teams for the best performance!
        \end{itemize}
      \end{itemize}
  \end{itemize}

  \begin{center}
    \textit{Registration is free, but mandatory, seats are limited!}\\
    \url{https://indico.math.cnrs.fr/e/kokkos-hackathon-26}
  \end{center}
